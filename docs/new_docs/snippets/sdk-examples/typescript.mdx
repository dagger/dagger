```typescript TypeScript icon="js"
import { connect } from "@dagger.io/dagger"

// Initialize Dagger client
connect(async (client) => {
  // Use Dagger client
  const output = await client
    .container()
    .from("node:18")
    .withExec(["node", "--version"])
    .stdout()
    
  console.log(output)
})
```
